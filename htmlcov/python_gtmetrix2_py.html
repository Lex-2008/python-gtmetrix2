46 def _plain_request(self, url, **kwargs): 

47 # method=None, data=None, headers={}, valid_status=None, valid_statuses=None, require_data=True): 

48 data=kwargs.get('data', None) 

49 if isinstance(data, dict): 

50 data = json.dumps(data) 

51 if isinstance(data, str): 

52 data = data.encode() 

53 request = urllib.request.Request( 

54 self.base_url + url, 

55 data=data, 

56 headers=kwargs.get('headers', {}), 

57 method=kwargs.get('method', None)) 

58 try: 

59 response = self._opener.open(request) 

60 except urllib.error.HTTPError as e: 

61 # https://docs.python.org/3/library/urllib.error.html#urllib.error.HTTPError 

62 # > Though being an exception (a subclass of URLError), an HTTPError 

63 # > can also function as a non-exceptional file-like return value 

64 # > (the same thing that urlopen() returns).  

65 # first seen at https://stackoverflow.com/a/52086806 

66 response = e 

67 if __debug__: 

68 # NOTE: in production, we don't check for status returned by API. 

69 # However, it might be helpful in CI tests 

70 if 'valid_status' in kwargs: 

71 kwargs['valid_statuses'] = [kwargs['valid_status']] 

72 if 'valid_statuses' in kwargs and response.status not in kwargs['valid_statuses']: 

73 raise GTmetrixAPIFailureException(('API returned invalid status: %s' % response.status), 

74 request, response, response.read()) 

75 data = response.read() 

76 if len(data) == 0: 

77 if kwargs.get('require_data', True): 

78 raise GTmetrixAPIFailureException('API returned empty response', request, response, data) 

79 else: 

80 return (response, None) 

81 try: 

82 json_data = json.loads(data.decode()) 

83 except (json.JSONDecodeError, UnicodeError) as e: 

84 raise GTmetrixAPIFailureException('API returned unparsable JSON', request, response, data) from e 

85 if 'errors' in json_data: 

86 if __debug__: 

87 if not isinstance(json_data['errors'], list): 

88 raise GTmetrixAPIFailureException('API returned non-list of errors', request, response, json_data) 

89 if len(json_data['errors']) < 1: 

90 raise GTmetrixAPIFailureException('API returned empty list of errors', request, response, json_data) 

91 if not all( (dict_is_error(x) for x in json_data['errors']) ): 

92 raise GTmetrixAPIFailureException('API returned non-error in error list', request, response, json_data) 

93 raise GTmetrixAPIErrorException(request, response, json_data) 

94 return (response, json_data)

96 def _retry_request(self, url, retries=10, **kwargs): 

97 # rest of arguments are same as for _plain request 

98 try: 

99 return self._plain_request(url, **kwargs) 

100 except GTmetrixAPIErrorException as e: 

101 for error in e.data['errors']: 

102 if error['status'] != '429': 

103 # we're interested only in 429 

104 raise 

105 if error['code'] == 'E42900': 

106 # Too many tests pending (Please retry after your existing tests have completed) 

107 # The recommended poll interval is 3 seconds. 

108 delay = 3 

109 elif error['code'] == 'E42901': 

110 # Rate limit exceeded 

111 # X-RateLimit-Reset contains # of Seconds remaining until the end of the current window 

112 delay = e.response.getheader('X-RateLimit-Reset', 3) 

113 delay = max(1, int(delay)) 

114 # TODO: error code E42901 but no ratelimit header - raise some exception instead of defaulting to 3? 

115 else: 

116 raise 

117 if retries <= 0: 

118 raise 

119 self._sleep(delay) 

120 return self._retry_request(url, retries-1, **kwargs)

164 def __init__(self, requestor, data, sleep_function=time.sleep): 

165 super().__init__(requestor, data, sleep_function)

168 def fetch(self, wait_for_complete=False, retries=10): 

169 (response, response_data) = self._requestor.request('tests/' + self['id']) 

170 if __debug__: 

171 if not 'data' in response_data: 

172 raise GTmetrixAPIFailureException('API returned no data for a test', None, response, response_data, self) 

173 if not dict_is_test(response_data['data']): 

174 raise GTmetrixAPIFailureException('API returned non-test for a test', None, response, response_data, self) 

175 self.update(response_data['data']) 

176 delay = response.getheader('Retry-After') 

177 if (not wait_for_complete or delay is None or retries <= 0): 

178 return 

179 delay = max(1, int(delay)) 

180 self._sleep(delay) 

181 return self.fetch(wait_for_complete, retries-1)
